  <script>
    emailjs.init('HqMt9o6fQRy4Ibt8h');

    function openModal(perfumeName){
      const modal = document.getElementById('orderModal');
      modal.classList.remove('hidden');
      modal.querySelector('#modalTitle').innerText = `Order ${perfumeName}`;
      modal.querySelector('.modal-show').classList.add('scaleIn');
    }

    function closeModal(){
      document.getElementById('orderModal').classList.add('hidden');
      document.getElementById('orderForm').reset();
    }

    document.getElementById('orderForm').addEventListener('submit', function(e){
      e.preventDefault();
      const form = e.target;
      const perfumeName = document.getElementById('modalTitle').innerText.replace('Order ','');
      const templateParams = {
        customer_name: form.customer_name.value,
        customer_email: form.customer_email.value,
        customer_phone: form.customer_phone.value,
        customer_address: form.customer_address.value,
        perfume_name: perfumeName,
        quantity: form.quantity.value
      };

      emailjs.send('service_86wyt9t','template_mksiuse', templateParams)
        .then(function(response){
          alert(`Order placed! ${perfumeName} x${form.quantity.value}`);
          closeModal();
        }, function(error){
          alert('Error sending order. Please try again.');
        });
    });
  </script>
